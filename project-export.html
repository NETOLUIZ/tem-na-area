// Formulrio final do cliente para criao do pedido.
import { useMemo, useState } from "react";
import { Link, useLocation, useNavigate } from "react-router-dom";
import { useApp } from "../store/AppContext";
import { formatCurrency } from "../utils/format";

export default function CheckoutPage() {
  const navigate = useNavigate();
  const location = useLocation();
  const { selectors, actions } = useApp();
  const cart = selectors.cartDetailed();

  const [form, setForm] = useState({
    nome: "",
    telefone: "",
    enderecoEntrega: location.state.enderecoEntrega || "",
    observacoes: ""
  });

  const [error, setError] = useState("");

  const totalItems = useMemo(
    () => cart.items.reduce((sum, row) => sum + row.quantidade, 0),
    [cart.items]
  );

  function validate() {
    if (form.nome.trim().length < 3) return "Informe o nome completo.";
    if (form.telefone.replace(/\D/g, "").length < 10) return "Telefone invlido.";
    if (form.enderecoEntrega.trim().length < 8) return "Informe o endereo de entrega.";
    return "";
  }

  function submit(e) {
    e.preventDefault();
    const message = validate();
    if (message) {
      setError(message);
      return;
    }

    const order = actions.createOrder(form);
    if (!order) {
      setError("No foi possvel finalizar. Verifique sua sacola.");
      return;
    }

    navigate("/pedido/sucesso", { state: { orderId: order.id } });
  }

  if (!cart.items.length) {
    return (
      <div className="container page-space">
        <div className="empty-state">
          <h3>Sacola vazia</h3>
          <p>Voc precisa adicionar itens antes de finalizar.</p>
          <Link className="btn btn-primary" to="/">Voltar para Home</Link>
        </div>
      </div>
    );
  }

  return (
    <main className="checkout-v2-page">
      <div className="checkout-v2-statusbar">
        <span>9:41</span>
        <div><span></span><span></span><span></span></div>
      </div>

      <header className="checkout-v2-header">
        <h1>Finalizar pedido</h1>
        <p>{cart.store.nome}</p>
      </header>

      <section className="checkout-v2-body">
        <form className="checkout-v2-form" onSubmit={submit} id="checkout-form">
          <div className="checkout-v2-card">
            <h2>Dados para entrega</h2>

            <input
              placeholder="Nome completo"
              value={form.nome}
              onChange={(e) => setForm({ ...form, nome: e.target.value })}
            />

            <input
              placeholder="Telefone"
              value={form.telefone}
              onChange={(e) => setForm({ ...form, telefone: e.target.value })}
            />

            <input

function getRelativeTime(value) {
  const date = new Date(value).getTime();
  const diff = Math.max(0, Date.now() - date);
  const min = Math.floor(diff / 60000);

  if (min < 1) return "Agora mesmo";
  if (min < 60) return `H ${min} min`;

  const hours = Math.floor(min / 60);
  if (hours < 24) return `H ${hours} h`;

  const days = Math.floor(hours / 24);
  return `H ${days} dia${days > 1  "s" : ""}`;
}

function getNextStatus(status) {
  if (status === "NOVO") return "PREPARO";
  if (status === "PREPARO") return "SAIU";
  if (status === "SAIU") return "CONCLUIDO";
  return status;
}

function getActionLabel(status) {
  if (status === "NOVO") return "Aceitar";
  if (status === "PREPARO") return "Enviar";
  if (status === "SAIU") return "Concluir";

}

      if (!Number.isInteger(value) || value < 1 || value > 5) return;

      setState((prev) => ({
        ...prev,
        stores: prev.stores.map((store) => {
          if (store.id !== storeId) return store;

          const currentRating = store.rating || { total: 0, count: 0, average: 0 };
          const total = currentRating.total + value;
          const count = currentRating.count + 1;

          return {
            ...store,
            rating: {
              total,
              count,
              average: total / count
            }
          };
        })
      }));
    },

    addToCart(storeId, itemId) {
      setState((prev) => {
        const incomingStore = prev.stores.find((store) => store.id === storeId);
        if (!incomingStore || incomingStore.status !== "ATIVA") return prev;

        const isSameStore = !prev.cart.storeId || prev.cart.storeId === storeId;
        const nextItems = isSameStore  [...prev.cart.items] : [];

        const found = nextItems.find((item) => item.itemId === itemId);
        if (found) {
          found.quantidade += 1;
        } else {
          nextItems.push({ itemId, quantidade: 1 });
        }

        return {
          ...prev,
          cart: {
            storeId,
            items: nextItems
          }
        };
      });
    },

          ...prev,
          cart: {
            storeId: next.length  prev.cart.storeId : null,
            items: next
          }
        };
      });

              nome: product.nome,
              preco: product.preco,
              quantidade: cartItem.quantidade

        const newPromotion = {
          id: uid("promo"),
          storeId,
          itemId: payload.itemId,
          title: payload.title,
          subtitle: payload.subtitle,
          badge: payload.badge,
          active: payload.active,
          createdAt,
          expiresAt
        };

        result = { ok: true, message: "Propaganda publicada. Ela ficara ativa por 48 horas." };

        return {
          ...prev,
          homePromotions: [newPromotion, ...(prev.homePromotions || [])]
        };
      });

      return result;
    },

    deleteHomePromotion(promotionId) {
      setState((prev) => ({
        ...prev,
        homePromotions: (prev.homePromotions || []).filter((promotion) => promotion.id !== promotionId)
      }));
    },

    upsertMenuItem(storeId, payload, itemId = null) {
      setState((prev) => {
        const current = prev.items.filter((item) => item.storeId === storeId);

        if (itemId) {
          return {
            ...prev,
            items: prev.items.map((item) =>
              item.id === itemId
                 {
                    ...item,
                    ...payload
                  }
                : item
            )
          };
        }

        const newItem = {
          id: uid("i"),
          storeId,
          nome: payload.nome,
          descricao: payload.descricao,
          preco: Number(payload.preco),
          precoAntigo: payload.precoAntigo  Number(payload.precoAntigo) : null,
          imagem: payload.imagem,
          tags: payload.tags || [],
          ativo: payload.ativo,
          createdAt: new Date().toISOString()
        };

        return {
          ...prev,
          items: [newItem, ...prev.items]
        };
      });
    },

    deleteMenuItem(itemId) {
      setState((prev) => ({
        ...prev,
        items: prev.items.filter((item) => item.id !== itemId)
      }));
    },

    updateOrderStatus(storeId, orderId, nextStatus) {
      if (!STATUS_FLOW.includes(nextStatus)) return;
      setState((prev) => ({
        ...prev,
        orders: prev.orders.map((order) =>
          order.id === orderId && order.storeId === storeId
             {
                ...order,
                status: nextStatus
              }
            : order
        )
      }));
    },

    loginSuperAdmin(user, pass) {
      const ok = user === "admin" && pass === "admin";
      if (!ok) return false;

      setState((prev) => ({
        ...prev,
        sessions: {
          ...prev.sessions,
          superAdmin: true
        }
      }));
      return true;
    },

    logoutSuperAdmin() {
      setState((prev) => ({
        ...prev,
        sessions: {
          ...prev.sessions,
          superAdmin: false
        }
      }));
    },

    superAdminAction(actionType, storeId, motivo = "") {
      setState((prev) => {
        const target = prev.stores.find((store) => store.id === storeId);
        if (!target) return prev;

        let stores = prev.stores;

        if (actionType === "APROVAR") {
          stores = stores.map((store) => (store.id === storeId  { ...store, status: "ATIVA" } : store));
        }

        if (actionType === "BLOQUEAR") {
          stores = stores.map((store) => (store.id === storeId  { ...store, status: "BLOQUEADA" } : store));
        }

        if (actionType === "DESBLOQUEAR") {
          stores = stores.map((store) => (store.id === storeId  { ...store, status: "ATIVA" } : store));
        }

        if (actionType === "EXCLUIR") {
          stores = stores.filter((store) => store.id !== storeId);
        }

        const nextOrders = actionType === "EXCLUIR"  prev.orders.filter((order) => order.storeId !== storeId) : prev.orders;
        const nextItems = actionType === "EXCLUIR"  prev.items.filter((item) => item.storeId !== storeId) : prev.items;

        const log = {
          id: `LOG-${prev.meta.logSequence + 1}`,
          actionType,
          storeId,
          storeName: target.nome,
          motivo: motivo || "Sem motivo informado",
          createdAt: new Date().toISOString()
        };

        return {
          ...prev,
          stores,
          items: nextItems,
          orders: nextOrders,
          logs: [log, ...prev.logs],
          meta: {
            ...prev.meta,
            logSequence: prev.meta.logSequence + 1
          }
        };
      });
    }
  }), [state.stores]);

  const selectors = useMemo(() => ({
    activeStores: state.stores.filter((store) => store.status === "ATIVA"),
    storeBySlug(slug) {
      return state.stores.find((store) => store.slug === slug) || null;
    },
    itemsByStore(storeId) {
      return state.items.filter((item) => item.storeId === storeId && item.ativo);
    },
    allItemsByStore(storeId) {
      return state.items.filter((item) => item.storeId === storeId);
    },
    homePromotionsByStore(storeId) {
      return (state.homePromotions || [])
        .filter((promotion) => promotion.storeId === storeId)
        .map((promotion) => {
          const item = state.items.find((entry) => entry.id === promotion.itemId) || null;
          return {
            ...promotion,
            item
          };
        });
    },
    activeHomePromotions() {
      const now = Date.now();
      return (state.homePromotions || [])
        .filter((promotion) => promotion.active && new Date(promotion.expiresAt).getTime() > now)
        .map((promotion) => {
          const store = state.stores.find((entry) => entry.id === promotion.storeId && entry.status === "ATIVA");
          const item = state.items.find((entry) => entry.id === promotion.itemId && entry.ativo);
          if (!store || !item) return null;

          return {
            ...promotion,
            store,
            item
          };
        })
        .filter(Boolean)
        .sort((a, b) => new Date(b.createdAt) - new Date(a.createdAt));
    },
    ordersByStore(storeId) {
      return state.orders.filter((order) => order.storeId === storeId);
    },
    cartDetailed() {
      const store = state.stores.find((item) => item.id === state.cart.storeId) || null;
      const items = state.cart.items
        .map((row) => {
          const item = state.items.find((product) => product.id === row.itemId);
          if (!item) return null;
          return {
            ...row,
            item,
            subtotal: item.preco * row.quantidade
          };
        })
        .filter(Boolean);

      const total = items.reduce((sum, row) => sum + row.subtotal, 0);

      return {
        store,
        items,
        total
      };
    }
  }), [state]);

  const value = {
    state,
    actions,
    selectors
  };

  return <AppContext.Provider value={value}>{children}</AppContext.Provider>;
}

export function useApp() {
  const context = useContext(AppContext);
  if (!context) throw new Error("useApp deve ser usado dentro de AppProvider");
  return context;

function ensureStoreRating(store) {
  const rating = store.rating || {};
  const total = Number.isFinite(rating.total)  rating.total : 0;
  const count = Number.isFinite(rating.count)  rating.count : 0;

  return {
    ...store,
    rating: {
      total,
      count,
      average: count > 0  total / count : 0
    }
  };
}

export function loadState() {
  try {
    const raw = localStorage.getItem(APP_STATE_KEY);
    if (!raw) return null;
    return JSON.parse(raw);
  } catch {
    return null;
  }
}

export function saveState(state) {
  localStorage.setItem(APP_STATE_KEY, JSON.stringify(state));
}

export function initState() {
  const current = loadState();
  if (current) {
    const seeded = createSeedState();

    return {
      ...current,
      stores: (current.stores || []).map(ensureStoreRating),
      homePromotions: Array.isArray(current.homePromotions)
         current.homePromotions.map(ensurePromotion)
        : seeded.homePromotions.map(ensurePromotion),
      contactLeads: current.contactLeads || []
    };
  }
  const seeded = createSeedState();
  saveState(seeded);
  return seeded;
}

export function clearState() {
  localStorage.removeItem(APP_STATE_KEY);
